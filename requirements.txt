# Core debugging / profiling toolset for this repo.
#
# Everything below is third-party; the standard library tools used
# throughout (traceback, faulthandler, pdb, gc, tracemalloc, cProfile,
# pstats, timeit, threading, multiprocessing, signal, dis) need no install.
#
# Install with:
#   python -m venv .venv
#   source .venv/bin/activate
#   pip install -r requirements.txt
#
# Tested on Python 3.14 (CPython, Linux x86_64). All of these ship
# Python 3.14 wheels as of mid-2026.

# --- Stack dumps -------------------------------------------------------
py-spy==0.4.2          # sampling profiler + "dump"/"top" for live processes
pystack==1.4.1         # stack dumps for live processes AND core dumps, with
                        # native (C/C++) frames + GIL/GC state. Linux only.
                        # See 01_stack_dumps/08_pystack.md
austin-python==2.2.0   # Python bindings/CLI helpers for the Austin sampling profiler
                        # (the `austin` binary itself is a separate native install,
                        # see 02_cpu_profiling/README.md)

# --- CPU profiling -------------------------------------------------------
line_profiler==5.0.2   # line-by-line CPU profiling via @profile decorator
pyinstrument==5.1.2    # low-overhead statistical call-stack profiler
scalene==2.3.0         # CPU + GPU + memory profiler, line-level, AI-assisted

# --- Memory profiling ------------------------------------------------------
memory-profiler==0.61.0  # line-by-line RSS memory profiling via @profile decorator
objgraph==3.6.2          # visualize / count object references (leak hunting)
Pympler==1.1             # heap analysis, object size + growth tracking
memray==1.19.3           # native + Python allocator memory profiler (Linux only)
psutil==7.2.2            # process-level CPU/memory/IO stats (used by memory_profiler)

# --- Docs ------------------------------------------------------------------
markdown-it-py==4.2.0    # renders this repo's .md files to HTML for `make docs`
